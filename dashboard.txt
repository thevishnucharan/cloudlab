<script>

const BASE_URL =
"https://naqjc0rrn2.execute-api.ap-south-1.amazonaws.com/prod";

window.onload = () => {

    loadAlerts();

    setInterval(loadAlerts, 60000);

};

async function checkWeather(){

    const btn=document.getElementById("checkBtn");

    const status=document.getElementById("status");

    const cards=document.getElementById("weatherCards");

    btn.disabled=true;

    btn.innerHTML="⏳ Checking...";

    status.className="loading";

    status.innerHTML=
    "Checking weather...";

    cards.innerHTML="";

    try{

        const response=await fetch(

            `${BASE_URL}/check`,

            {

                method:"POST"

            }

        );

        const data=await response.json();

        if(data.results){

            data.results.forEach(city=>{

                if(city.error){

                    return;

                }

                const div=document.createElement("div");

                div.className="card";

                if(city.alertTriggered){

                    div.classList.add("alert-card");

                }

                div.innerHTML=`

                    <h3>${city.city}</h3>

                    <div class="temp">

                        ${city.temperature}°C

                    </div>

                    <div class="desc">

                        ${city.description}

                    </div>

                    <div class="hum">

                        Humidity :
                        ${city.humidity}%

                    </div>

                    <span class="badge
                    ${city.alertTriggered ? "alerted" : "normal"}">

                    ${city.alertTriggered ?

                    "🚨 Alert Sent"

                    :

                    "✅ Safe"}

                    </span>

                `;

                cards.appendChild(div);

            });

        }

        if(data.alertsTriggered>0){

            status.className="error";

            status.innerHTML=

            `🚨 ${data.alertsTriggered}
            alert(s) triggered.
            Email notification sent successfully.`;

        }

        else{

            status.className="success";

            status.innerHTML=

            "✅ Weather checked successfully. No disaster detected.";

        }

        loadAlerts();

    }

    catch(error){

        console.log(error);

        status.className="error";

        status.innerHTML=

        "❌ Error while checking weather.";

    }

    btn.disabled=false;

    btn.innerHTML="⚡ Check Weather Now";

}
async function loadAlerts(){

    try{

        const response=await fetch(

            `${BASE_URL}/alerts`

        );

        const data=await response.json();

        const tbody=document.getElementById("alertBody");

        if(!data.length){

            tbody.innerHTML=`

            <tr>

            <td colspan="6"
            class="no-data">

            No alerts have been generated yet.

            </td>

            </tr>

            `;

            return;

        }

        tbody.innerHTML="";

        data.forEach(alert=>{

            const row=document.createElement("tr");

            const date=new Date(

                Number(alert.timestamp)*1000

            ).toLocaleString();

            let sourceBadge="";

            if(alert.triggerSource==="manual"){

                sourceBadge=`

                <span class="source manual">

                ⚡ Manual

                </span>

                `;

            }

            else{

                sourceBadge=`

                <span class="source auto">

                ⏱ Automatic

                </span>

                `;

            }

            row.innerHTML=`

            <td>${alert.city}</td>

            <td>${alert.description}</td>

            <td>${alert.temperature}°C</td>

            <td>${alert.humidity}%</td>

            <td>${date}</td>

            <td>${sourceBadge}</td>

            `;

            tbody.appendChild(row);

        });

    }

    catch(error){

        console.log(error);

        document.getElementById("alertBody").innerHTML=`

        <tr>

        <td colspan="6"
        class="no-data">

        Unable to load alert history.

        </td>

        </tr>

        `;

    }

}
