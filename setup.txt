2.2 Add Email Subscription (Fix for Email Alerts Not Sending)
⚠  Why email was not working before
The most common reason email alerts do not arrive is that the SNS subscription was never confirmed.
AWS sends a confirmation email when you subscribe — you MUST click the link inside it.
Until confirmed, the subscription shows as PendingConfirmation and SNS skips it silently.
Follow steps below carefully and confirm within 48 hours (link expires).

import json
import boto3
import urllib.request
import time
import os
 
dynamodb = boto3.resource('dynamodb')
sns      = boto3.client('sns')
 
ALERTS_TABLE  = 'Alerts'
SNS_TOPIC_ARN = os.environ['SNS_TOPIC_ARN']
API_KEY       = os.environ['WEATHER_API_KEY']
CITIES        = ['Bengaluru', 'Mumbai', 'Delhi']
 
# ── Alert threshold (lower this to test; real value is 42) ──────────────
TEMP_HIGH = 30   # alert if temperature ABOVE this value in Celsius
TEMP_LOW  = 5    # alert if temperature BELOW this value in Celsius
 
 
def send_alert(table, city, description, temp, humidity, weather_id, source):
    """Save to DynamoDB and publish to SNS. Called for every triggered alert."""
    alert_id = f'{city}-{int(time.time())}'
    ts       = int(time.time())
 
    # ── Save to DynamoDB ─────────────────────────────────────────────────
    table.put_item(Item={
        'alertId':       alert_id,
        'timestamp':     ts,
        'city':          city,
        'description':   description,
        'temperature':   str(temp),
        'humidity':      str(humidity),
        'weatherCode':   weather_id,
        'triggerSource': source,   # 'auto' or 'manual'
    })
    print(f'DynamoDB write OK: {alert_id}')
 
    # ── Publish to SNS ───────────────────────────────────────────────────
    message = (
        f'DISASTER ALERT\n'
        f'City       : {city}\n'
        f'Condition  : {description.upper()}\n'
        f'Temperature: {temp}C\n'
        f'Humidity   : {humidity}%\n'
        f'Trigger    : {source.upper()}\n'
        f'Time       : {time.strftime("%Y-%m-%d %H:%M:%S UTC", time.gmtime(ts))}'
    )
    response = sns.publish(
        TopicArn = SNS_TOPIC_ARN,
        Message  = message,
        Subject  = f'Disaster Alert: {city} - {description}',
    )
    print(f'SNS publish OK: MessageId={response["MessageId"]}')
    return alert_id
 
 
def check_all_cities(source):
    """
    Fetch weather for every city, evaluate thresholds, send alerts.
    source: 'auto' (EventBridge) or 'manual' (dashboard button)
    """
    table        = dynamodb.Table(ALERTS_TABLE)
    alerts_sent  = []
    results      = []
 
    for city in CITIES:
        try:
            url = (
                'http://api.openweathermap.org/data/2.5/weather'
                f'?q={city}&appid={API_KEY}&units=metric'
            )
            print(f'Fetching weather for {city}...')
            with urllib.request.urlopen(url, timeout=10) as resp:
                data = json.loads(resp.read())
 
            weather_id  = data['weather'][0]['id']
            description = data['weather'][0]['description']
            temp        = data['main']['temp']
            humidity    = data['main']['humidity']
            print(f'{city}: {description}, {temp}C, humidity={humidity}%, code={weather_id}')
 
            # ── Evaluate thresholds ──────────────────────────────────────
            thunderstorm   = (200 <= weather_id <= 299)
            extreme_event  = (weather_id >= 900)
            heat_alert     = (temp > TEMP_HIGH)
            cold_alert     = (temp < TEMP_LOW)
            triggered      = thunderstorm or extreme_event or heat_alert or cold_alert
 
            print(f'  thunderstorm={thunderstorm}, extreme={extreme_event}, '
                  f'heat={heat_alert}, cold={cold_alert} => triggered={triggered}')
 
            alert_id = None
            if triggered:
                alert_id = send_alert(
                    table, city, description, temp, humidity, weather_id, source
                )
                alerts_sent.append(alert_id)
 
            results.append({
                'city':           city,
                'description':    description,
                'temperature':    temp,
                'humidity':       humidity,
                'weatherCode':    weather_id,
                'alertTriggered': triggered,
                'alertId':        alert_id,
            })
 
        except Exception as e:
            print(f'ERROR for {city}: {e}')
            results.append({'city': city, 'error': str(e)})
 
    return {
        'alertsTriggered': len(alerts_sent),
        'citiesChecked':   len(results),
        'results':         results,
        'source':          source,
        'checkedAt':       int(time.time()),
    }
 
 
def cors_headers():
    return {
        'Access-Control-Allow-Origin':  '*',
        'Access-Control-Allow-Methods': 'GET,POST,OPTIONS',
        'Access-Control-Allow-Headers': 'Content-Type',
        'Content-Type':                 'application/json',
    }
 
 
def lambda_handler(event, context):
    http_method = event.get('httpMethod', '')
    path        = event.get('path', '')
 
    # ── TRIGGER 1: EventBridge scheduled run ──────────────────────────
    if event.get('source') == 'aws.events':
        print('=== SCHEDULED RUN (EventBridge) ===')
        summary = check_all_cities('auto')
        print(f'Scheduled run done: {summary["alertsTriggered"]} alerts')
        return {'statusCode': 200, 'body': json.dumps(summary)}
 
    # ── TRIGGER 2: CORS preflight ─────────────────────────────────────
    if http_method == 'OPTIONS':
        return {'statusCode': 200, 'headers': cors_headers(), 'body': ''}
 
    # ── TRIGGER 3: Manual check from dashboard button ─────────────────
    if http_method == 'POST' and path == '/check':
        print('=== MANUAL CHECK (Dashboard button) ===')
        summary = check_all_cities('manual')
        return {
            'statusCode': 200,
            'headers':   cors_headers(),
            'body':      json.dumps({
                'message':         f'Checked {summary["citiesChecked"]} cities',
                'alertsTriggered': summary['alertsTriggered'],
                'results':         summary['results'],
                'checkedAt':       summary['checkedAt'],
            }),
        }
 
    # ── TRIGGER 4: Dashboard reads alert history ──────────────────────
    if http_method == 'GET' and path == '/alerts':
        print('=== GET /alerts (Dashboard read) ===')
        table  = dynamodb.Table(ALERTS_TABLE)
        result = table.scan(Limit=50)
        items  = sorted(
            result['Items'],
            key=lambda x: int(x['timestamp']),
            reverse=True
        )
        return {
            'statusCode': 200,
            'headers':   cors_headers(),
            'body':      json.dumps(items),
        }
 
    # ── Fallback ──────────────────────────────────────────────────────
    print(f'Unknown trigger: method={http_method}, path={path}')
    return {
        'statusCode': 400,
        'headers':   cors_headers(),
        'body':      json.dumps({'error': 'Unknown request type'}),
    }

    const BASE = 'https://YOUR_INVOKE_URL/prod';

    async function checkNow() {
      const btn = document.getElementById('checkBtn');
      const st  = document.getElementById('status');
      const cards = document.getElementById('cards');
      btn.disabled = true; btn.textContent = 'Checking...';
      st.className = 'loading';
      st.textContent = 'Lambda running... fetching weather for all cities...';
      cards.innerHTML = '';
      try {
        const res  = await fetch(`${BASE}/check`, {method:'POST'});
        const data = await res.json();
        (data.results||[]).forEach(r => {
          if (r.error) return;
          const d = document.createElement('div');
          d.className = 'card' + (r.alertTriggered?' alerted':'');
          d.innerHTML = `<h3>${r.city}</h3>
            <div class='tmp'>${r.temperature}C</div>
            <div class='dsc'>${r.description}</div>
            <div class='hum'>Humidity: ${r.humidity}%</div>
            <span class='badge ${r.alertTriggered?'bad':'ok'}'>
              ${r.alertTriggered?'ALERT SENT':'Normal'}</span>`;
          cards.appendChild(d);
        });
        const n = data.alertsTriggered||0;
        st.className = n>0?'err':'ok';
        st.textContent = n>0
          ? `${n} alert(s) triggered - SMS and email sent`
          : 'All cities checked - no dangerous conditions';
        await loadAlerts();
      } catch(e) {
        st.className='err'; st.textContent='Error: '+e.message;
      }
      btn.disabled=false; btn.textContent='Check Weather Now';
    }
 
    async function loadAlerts() {
      try {
        const res  = await fetch(`${BASE}/alerts`);
        const data = await res.json();
        const tb   = document.getElementById('tbody');
        if (!data.length) {
          tb.innerHTML = '<tr><td colspan=6 class=empty>No alerts yet</td></tr>';
          return;
        }
        tb.innerHTML = data.map(a => `<tr>
          <td><strong>${a.city}</strong></td>
          <td>${a.description}</td>
          <td>${a.temperature}C</td>
          <td>${a.humidity||'-'}%</td>
          <td>${new Date(a.timestamp*1000).toLocaleString()}</td>
          <td><span class='src ${a.triggerSource||'auto'}'>
            ${a.triggerSource==='manual'?'Manual':'Auto'}</span></td>
        </tr>`).join('');
      } catch(e) { console.error(e); }
    }
